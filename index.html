// 💡 두 좌표 사이의 거리를 계산하는 도우미 함수 (km 단위) function getDistance(lat1, lon1, lat2, lon2) { const R = 6371; const dLat = (lat2 - lat1) * Math.PI / 180; const dLon = (lon2 - lon1) * Math.PI / 180; const a = Math.sin(dLat / 2) * Math.sin(dLat / 2) + Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * Math.sin(dLon / 2) * Math.sin(dLon / 2); const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a)); return R * c; } async function searchPole() { var poleNum = document.getElementById('pole-input').value.toUpperCase().trim(); if (poleNum.length !== 8) { alert("인입주번호 8자리를 입력해주세요. (예: 1234A567)"); return; } var statusEl = document.getElementById('status'); statusEl.innerText = "검색 중..."; var fileKey = poleNum.substring(0, 4); var url = `https://raw.githubusercontent.com/${GITHUB_USER}/${GITHUB_REPO}/${GITHUB_BRANCH}/${fileKey}.csv`; try { const response = await fetch(url); if (!response.ok) { alert("해당 데이터 파일을 찾을 수 없습니다. (" + fileKey + ".csv)"); statusEl.innerText = "조회 실패"; return; } const text = await response.text(); const lines = text.trim().split('\n'); var foundList = []; for (var i = 1; i < lines.length; i++) { var cols = lines[i].split(','); if (cols[0] && cols[0].trim().toUpperCase() === poleNum) { foundList.push({ meshno: cols[0].trim(), addr: cols[1] ? cols[1].trim() : "주소 정보 없음", lng: parseFloat(cols[2]), lat: parseFloat(cols[3]) }); } } if (foundList.length > 0) { // 💡 [핵심 로직] 기준점 결정 (내 위치 > 마지막 마커 순서) var refPoint = null; if (myLocMarker) { // 1순위: GPS 위치 추적 중이면 내 현재 위치 기준 refPoint = { lat: myLocMarker.getPosition().getLat(), lng: myLocMarker.getPosition().getLng() }; } else if (currentSites.length > 0) { // 2순위: GPS는 없지만 지도에 찍힌 마커가 있다면 가장 최근 마커 기준 var lastSite = currentSites[currentSites.length - 1]; refPoint = { lat: lastSite.lat, lng: lastSite.lng }; } // 💡 기준점이 존재할 때만 '가장 가까운 1곳'만 남기기 if (refPoint) { foundList.sort(function(a, b) { var distA = getDistance(refPoint.lat, refPoint.lng, a.lat, a.lng); var distB = getDistance(refPoint.lat, refPoint.lng, b.lat, b.lng); return distA - distB; // 오름차순 정렬 (가까운 순) }); // 가장 가까운 딱 1개만 선택 foundList = [foundList[0]]; } // 마커 표시 프로세스 foundList.forEach(function(found) { var poleSite = { id: crypto.randomUUID(), title: "전주: " + found.meshno, lat: found.lat, lng: found.lng, addr: found.addr, colorCode: 7, isAbsent: false, memo: "GitHub 데이터 확인 완료" }; currentSites.push(poleSite); }); saveAndRender(); map.setCenter(new kakao.maps.LatLng(foundList[0].lat, foundList[0].lng)); map.setLevel(2); // 현장이 잘 보이도록 확대 if (refPoint) { statusEl.innerText = "현재 위치 기반 최단거리 조회 성공!"; } else { statusEl.innerText = "전국 중복 데이터 " + foundList.length + "건 모두 표시"; alert("내 위치 정보나 기준 마커가 없어 검색된 " + foundList.length + "건을 모두 지도에 표시했습니다."); } } else { alert("입력하신 번호 (" + poleNum + ")를 찾을 수 없습니다."); statusEl.innerText = "조회 실패"; } } catch (error) { console.error("조회 에러:", error); alert("데이터 조회에 실패했습니다."); statusEl.innerText = "오류 발생"; } }
